```
import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
		square sq = new square();
		sq.x1 = sc.nextInt();
		sq.y1 = sc.nextInt();
		sq.x2 = sc.nextInt();
		sq.y2 = sc.nextInt();
		sq.x3 = sc.nextInt();
		sq.y3 = sc.nextInt();
		sq.x4 = sc.nextInt();
		sq.y4 = sc.nextInt();
		
		sq.cal();
		
		System.out.print(sq.nx1+" ");
		System.out.print(sq.ny1+" ");
		System.out.print(sq.nx2+" ");
		System.out.print(sq.ny2+" ");
	}

}

class square {
	static int x1, y1; // 1번 사각형 왼쪽아래 좌표
	static int x2, y2; // 1번 사각형 오른쪽 위 좌표
	static int x3, y3; // 2번 사각형 왼쪽아래 좌표
	static int x4, y4; // 2번 사각형 오른쪽 위 좌표

	static int nx1, ny1, nx2, ny2; // 최종 사각형 좌표

	square() {
	}

	static void cal() 
	{
	if(x1<x3) {
		nx1= x1;
	} else {
		nx1= x3;
	}
	if(x2<x4) {
		nx2= x4;
	} else {
		nx2= x2;
	}
	if(y1<y3) {
		ny1= y1;
	} else {
		ny1= y3;
	}
	if(y2<y4) {
		ny2= y4;
	} else {
		ny2= y2;
	}
}
}
```
